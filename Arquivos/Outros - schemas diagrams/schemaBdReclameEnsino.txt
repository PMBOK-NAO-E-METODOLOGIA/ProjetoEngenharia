CREATE DATABASE ReclameEnsinoBD;

CREATE TABLE tb_administradores(
	idAdmin			   INT PRIMARY KEY IDENTITY,
	nomeAdmin          VARCHAR(50) NOT NULL,
	nomeUsuario        VARCHAR(20) UNIQUE NOT NULL,
	senhaUsuario	   VARCHAR(20) NOT NULL,
	dataCadastro       DATE NOT NULL,
	isAtivo		       CHAR(1),
	CONSTRAINT CHK_isAtivo CHECK(isAtivo = 'S' OR isAtivo = 'N');
)

CREATE TABLE comentarios(
	id				 INT PRIMARY KEY IDENTITY, --VERIFICAR COM A GALERA A EXISTENCIA DESTE CAMPO ASSIM COMO OS DEMAIS
	nome	   		 VARCHAR(50),
	comentario 		 VARCHAR(1000) NOT NULL,   --VERIFICAR A QUANTIDADE DE CARACTERES
	dataDoComentario DATE NOT NULL,
	foiVisto		 BIT --
)

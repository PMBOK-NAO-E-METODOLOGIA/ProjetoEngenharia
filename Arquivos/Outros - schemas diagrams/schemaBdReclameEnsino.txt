CREATE DATABASE reclameEnsino;

CREATE TABLE admins(
	id			    INT PRIMARY KEY IDENTITY,
	nome	        VARCHAR(50) NOT NULL,
	nomeDeUsuario   VARCHAR(20) NOT NULL,
	senha		    VARCHAR(20) NOT NULL,
	dataDoCadastro  DATETIME NOT NULL,
	estaAtivo		BOOLEAN --
)

CREATE TABLE comentarios(
	id				 INT PRIMARY KEY IDENTITY, --VERIFICAR COM A GALERA A EXISTENCIA DESTE CAMPO ASSIM COMO OS DEMAIS
	nome	   		 VARCHAR(50),
	comentario 		 VARCHAR(1000) NOT NULL,   --VERIFICAR A QUANTIDADE DE CARACTERES
	dataDoComentario DATETIME NOT NULL,
	foiVisto		 BOOLEAN --
)
